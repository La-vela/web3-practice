## NusaPayPayroll.sol
급여 자동화 스마트 컨트랙트
- 예외구문을 가드레일로 require를 대부분 사용 (보안 최적화)
-
```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

// (1) 외부 라이브러리: 안전한 달러 코인(USDC) 전송을 위한 표준 규약
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";

contract NusaPayPayroll {
    IERC20 public usdcToken;
    address public companyOwner; // 회사 대표(관리자) 지갑 주소

    // (2) 데이터 모델: 직원의 월급 정보를 담는 명세서 
    struct Employee {
        address walletAddress; // 직원의 코인 지갑 주소
        uint256 salaryAmount;  // 월급 (USDC 기준)
        string fiatCurrency;   // 최종적으로 받고 싶은 나라 돈 (예: "IDRX", "KRW")
        bool isActive;         // 현재 재직 중인지 여부 (퇴사자 방지)
    }

    // 직원 ID(사번)
    mapping(string => Employee) public employees; 

    // 영수증 발급기: 급여 지급 후 블록체인에 영구적으로 기록을 남김
    event PayrollProcessed(string employeeId, uint256 amount, string fiatCurrency);

    constructor(address _usdcAddress) {
        usdcToken = IERC20(_usdcAddress);
        companyOwner = msg.sender; // 컨트랙트를 배포한 사람을 사장님으로 등록
    }

    // (3) 급여 지급 로직: 조건문과 실행
    function payEmployee(string memory _employeeId) public {
        Employee memory emp = employees[_employeeId];
        
        // Guardrail 1: 현재 재직 중인 직원이 맞는지 검사
        require(emp.isActive == true, "Error: Not an active employee"); 
        
        // Guardrail 2: 금고(회사)에 직원 월급을 줄 만큼 돈이 충분한지 검사
        require(usdcToken.balanceOf(address(this)) >= emp.salaryAmount, "Error: Insufficient company funds");

        // (4) 실제 송금 실행
        bool success = usdcToken.transfer(emp.walletAddress, emp.salaryAmount);
        require(success, "Error: Transfer failed"); // 송금 실패 시 거래를 아예 없던 일로 되돌림(Rollback)

        // (5) 투명한 기록 남기기
        emit PayrollProcessed(_employeeId, emp.salaryAmount, emp.fiatCurrency);
    }
}
```

## Crypto kitty란?
대중이 이해하기 어려운 블록체인(ERC-721) 기술을 '귀여운 고양이 수집 게임'으로 포장
- 고양이의 유전 정보를 uint256 (256자리의 거대한 이진수 숫자) 하나로 압축해서 저장
- 심리학에서 말하는 스키너의 상자, 즉 '변동 비율 강화(Variable-ratio schedule)'로 결과를 예측할 수 없는 기대감이 뇌에 도파민을 분비하게 함

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MiniCryptoKitties {
    
    // ==========================================
    // 1. 데이터 모델: 고양이의 실체 (DB 설계)
    // ==========================================
    struct Kitty {
        uint256 genes;      // 256비트 유전자 숫자 (핵심 데이터)
        uint64 birthTime;   // 탄생 시간
        uint32 matronId;    // 엄마 고양이의 ID 번호
        uint32 sireId;      // 아빠 고양이의 ID 번호
    }

    // 세상에 태어난 모든 고양이들을 순서대로 저장하는 거대한 배열
    Kitty[] public kitties;


    // ==========================================
    // 2. 소유권 장부: ERC-721의 핵심
    // ==========================================
    // "이 고양이 ID의 주인은 이 지갑 주소다"라고 기록하는 영구 장부
    mapping(uint256 => address) public kittyToOwner;


    // ==========================================
    // 3. 고양이 탄생 (Minting): 새 생명 창조 로직
    // ==========================================
    function _createKitty(
        uint32 _matronId, 
        uint32 _sireId, 
        uint256 _genes, 
        address _owner
    ) internal returns (uint256) {
        // 새로운 고양이 데이터 생성
        Kitty memory _kitty = Kitty({
            genes: _genes,
            birthTime: uint64(block.timestamp),
            matronId: _matronId,
            sireId: _sireId
        });

        // 고양이 배열에 추가하고, 부여받은 고유 ID 번호 확인
        kitties.push(_kitty);
        uint256 newKittenId = kitties.length - 1;

        // ⭐️ 소유권 장부에 주인의 이름(지갑 주소)을 도장 찍음
        kittyToOwner[newKittenId] = _owner;

        return newKittenId;
    }

    // ==========================================
    // 4. 교배 시스템 (돌연변이 로직 추가)
    // ==========================================
    function breed(uint256 _matronId, uint256 _sireId) public {
        
        require(kittyToOwner[_matronId] == msg.sender, "Not your mother cat");
        require(kittyToOwner[_sireId] == msg.sender, "Not your father cat");

        uint256 matronGenes = kitties[_matronId].genes;
        uint256 sireGenes = kitties[_sireId].genes;

        // 1단계: 기본 유전자 섞기 (부모의 유전자를 반반씩)
        uint256 childGenes = (matronGenes / 2) + (sireGenes / 2);

        // 2단계: 난수(랜덤 숫자) 생성하기 (0 ~ 99 사이의 숫자 뽑기)
        // 주의: 실제 서비스에서는 이 방식(block.timestamp)을 쓰면 해커가 예측할 수 있어 위험합니다!
        uint256 randomNum = uint256(keccak256(abi.encodePacked(block.timestamp, msg.sender))) % 100;

        // 3단계: 돌연변이 발생 조건문 (예: 5% 확률로 대박 터짐!)
        if (randomNum < 5) {
            // 부모에게 전혀 없는, 엄청나게 희귀한 '전설의 돌연변이 유전자' 부여!
            childGenes = 999999999999; 
        }

        // 새로운 아기 고양이 탄생
        _createKitty(uint32(_matronId), uint32(_sireId), childGenes, msg.sender);
    }
