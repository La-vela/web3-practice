## Crypto kitty
ERC-721(NFT) 데이터 기본 구조
- 공개글(public), 속성(struct), 장부(mapping)
- 경제적인 희소성 유지 - 비공개(private)

```
pragma solidity 0.8.17;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";

contract SimpleCryptoKitties is ERC721 {
    uint256 public _tokenIdCounter = 1;

  struct Kitty {
    uint256 genes;
    uint256 birthTime;
    uint256 momId;
    uint256 dadId;
    uint256 generation;
  }

  mapping(uint256 => Kitty) public kitties; // 고양이가 위치한 지갑
  mapping(uint256 => address) private _owners;  // 고양이 주인이 위치한 지갑

  constructor() ERC721("SimpleCryptoKitties", "SCK") {}

  function _createKitty(uint256 momId, uint256 dadId, uint256 generation, uint256genes, address owner) private returns (uint256) { }
}
```
