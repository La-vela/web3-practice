### Function Modifier (함수 수정자)

함수 수정자 정의
```
contract Example {
    uint public x;

    modifier checkValue(uint _value) {
        require(_value > x, "Value is not greater than x");
        _;
    }
}
```

책 판매 예제 코드 "누구든 100원 이상 들고 오면 살 수 있다"
```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract BookStore {

    // modifier가 하는 일: "이 조건을 통과해야 함수를 실행시켜줄게"
    modifier checkPrice() {
        require(msg.value >= 100, "100원 이상 필요합니다");
        _;  // 자리표시자 "여기에 원래 함수 코드 본문(ex. 실체 책 구매 로직)을 실행해라"
    }

    // 함수 뒤에 이름만 붙여서 연결 (내부 코드는 그대로)
    function buyBook() public payable checkPrice {
        // 실제 책 구매 로직
    }

    // 다른 함수에도 checkPrice 하나만 붙이면 동일하게 적용
    function buyPen() public payable checkPrice {
        // 펜 구매 로직
    }

    function buyEraser() public payable checkPrice {
        // 지우개 구매 로직
    }
}
```

### 뉴스 접근 권한 제어 컨트랙트 (뉴스 열람권)
실제 유료 NFT 멤버십 뉴스 구독 서비스의 핵심 로직
1. 뉴스사(News Provider)는 기사 ID와 가격을 컨트랙트에 등록
2. 에이전트(Buyer/Developer)는 소량의 토큰을 지불하고 purchaseArticle을 호출
3. 에이전트가 뉴스 API에 데이터를 요청하면, 서버의 온체인 상의 checkAccess 결과를 확인한 뒤 온톨로지 데이터를 반환

[특징]
- Access Control (접근 제어)
    누가(Address) 어떤 기사(ID)에 대한 열람권(Boolean)"을 가졌는지 0과 1로 기록
- ERC-20 결제 연동
    코드 내의 IERC20 인터페이스와 transferFrom 함수는 전 세계 모든 이더리움 기반 토큰(USDC, USDT 등)이 따르는 국제 표준
- L2(Layer 2) 최적화
    1장 단위의 소액 결제로 사용한다면 이더리움보단 Base나 Polygon이 적절

[추가 요소]
Ownable (관리자 권한)
Pull over Push (출금 방식)
실시간 환율 Oracle 연동 (Chainlink)


Solidity: 접근 권한 제어
- Backend API(온체인 검증 & 데이터 전달)와 Agent 온톨로지 변환 그래프 코드는 별도

```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";

contract NewsAccessControl {
    IERC20 public usdcToken;
    address public newsProvider;
    uint245 public pricePerArticle = 100000; // 0.1 USDC (6 decimals / 소수점 6자리)

    // [구매자][기사ID] => 구매 완료 여부
    mapping(address => mapping(string => bool)) public hasAccess;

    constructor(address _usdc, address _provider) {
        usdcToken = IERC20(_usdc);
        newsProvider = _provider;
    }

    // 기사 1건 구매
    function purchase(string memory articleId) external {
        require(!hasAccess[msg.sender][articleId], "Already purchased");

        // 결제 실행
        require(usdcToken.transferFrom(msg.sender, newsProvider, pricePerArticle), "Payment failed");

        // 권한 기록
        hasAccess[msg.sender][articleId] = true;
    }
}

```
