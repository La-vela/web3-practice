### ERC721 - NFT 제출 & 발행 & 조회, 관리자 승인 기능

- submitRecipe()

```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

contract RecipeContest {

    uint256 public nextRecipeId;

    enum Status {
        Pending,
        Approved,
        Rejected
    }

    struct Recipe {
        uint256 recipeId;
        address creator;
        string recipeName;
        string ingredients;
        uint256 score;
        Status status;
    }

    // recipeId => Recipe
    mapping(uint256 => Recipe) public recipes;

    // =================== NFT 개념 ====================
    // tokenId => owner
    mapping(uint256 => address) public owners;

    // tokenId => recipeId
    mapping(uint256 => uint256) public nftRecipes;

    uint256 public nextTokenId;

    address public admin;

    constructor() {
        admin = msg.sender;
    }

    // 레시피 제출
    function submitRecipe(
        string memory recipeName,
        string memory ingredients
    )
        public
    {
        nextRecipeId++;

        recipes[nextRecipeId] = Recipe({
            recipeId: nextRecipeId,
            creator: msg.sender,
            recipeName: recipeName,
            ingredients: ingredients,
            score: 0,
            status: Status.Pending
        });
    }

    // 관리자 승인
    function approveRecipe(
        uint256 recipeId,
        uint256 score
    )
        public
    {
        require(
            msg.sender == admin,
            "Only admin"
        );

        Recipe storage recipe =
            recipes[recipeId];

        recipe.status =
            Status.Approved;

        recipe.score =
            score;

        mintNFT(
            recipe.creator,
            recipeId
        );
    }

    // NFT 발행
    function mintNFT(
        address owner,
        uint256 recipeId
    )
        internal
    {
        nextTokenId++;

        owners[nextTokenId] =
            owner;

        nftRecipes[nextTokenId] =
            recipeId;
    }

    // NFT 주인 조회
    function ownerOf(
        uint256 tokenId
    )
        public
        view
        returns(address)
    {
        return owners[tokenId];
    }
}
```

### 사용자 레시피 호출(저장)
```solidity
submitRecipe(
    "Wasabi Burger",
    "Patty, Cheese, Wasabi"
);
```

결과:
``` 
- recipes 메모리 저장 구조
│
├─ 1
│   ├─ recipeName : Wasabi Burger
│   ├─ ingredients : Patty, Cheese, Wasabi
│   └─ status : Pending
```

### 관리자 레시피 승인

```solidity
approveRecipe(
    1,
    95
);
```


결과:
```
recipes
│
├─ 1
│   ├─ recipeName : Wasabi Burger
│   ├─ ingredients : Patty, Cheese, Wasabi
│   ├─ score : 95
│   └─ status : Approved
```

```text
owners

nftRecipes

```
