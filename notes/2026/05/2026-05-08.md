### interface (인터페이스)

```
interface interfaceA {
    function myFunction(uint256 x) external returns (uint256);
}


# 인터페이스 상속
// 인터페이스 상속은 상태 변수를 선언할 수 없음
interface interfaceB {
    function myFunction(uint256 x) external returns (uint256);
}

contract ContractB is InterfaceB {  // ContractB는 InterfaceB 규격에 따름
    uint256 private data;

    // Constructor to initialize data
    constructor() {
        data = 0;
    }

    // Implementing the doSomething function from InterfaceA
    function doSomething(uint256 value) external override returns (uint256) {
        data += value;
        return data;
    }
}

```
