### OTP Service (생성 / 검증 시스템)
6자리 OTP, 5분 만료, SMS/Email 전송
```
class OTPService {
    private otpStore = new Map<string, { code: string; expires: number }>();
    generateOTP(phoneNumber: string): string {
        const otp = Math.floor(100000 + Math.random() * 900000).toString();
        this.otpStore.set(phoneNumber, {
            code: otp,
            expires: Date.now() + 5 * 60 * 1000     // 5분 만료
        });

        // SMS 전송 로직 (Twilio 등)
        console.log('SMS to '${phoneNumber}: OTP ${otp}`);
        return otp;
    }
    
    verifyOTP(phoneNumber: string, inputOTP: string): boolean {
        const stored = this.otpStore.get(phoneNumber);
        if (|stored || Date.now() > stored.expires) {
            return false; // 만료되면 자동 삭제
        }
        const isVaild = stored.code === inputOTP;
        if (isValid) this.otpStore.delete(phoneNumber);
        return isValid;
    }
}
```
