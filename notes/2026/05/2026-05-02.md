## example

### whitelist (매표 할인)
```
// SPDX-Licencse-Identifier: MIT
pragma solidity ^0.8.17;

contract WhitelistMetaverse {
    address public owner;
    uint public constant PUBLIC_PRICE = 0.05 ether;     // 일반 가격
    uint public constant WHITELIST_PRICE = 0.01 ether;  // 화이트리스트(할인) 가격

    mapping(address => bool) public whitelist;          // 화이트리스트 명단
    mapping(address => bool) public hasAccess;

    constructor() {
        owner = msg.sender;
    }

    // [핵심] 관리자가 화이트리스트 명단을 추가하는 함수
    function addToWhitelist(address _user) public {
        require(msg.sender == owner, "Only owner can add");
        whitelist[_user] = true;
    }

    // [핵심] 화이트리스트 여부에 따라 가격을 다르게 체크
    function buyTicket() public payable {
        // 화이트리스트라면 할인 가격 적용, 아니면 일반 가격 적용
        uint requiredPrice = whitelist[msg.sender] ? WHITELIST_PRICE : PUBLIC_PRICE;

        require(msg.value >= requiredPrice, "Sent Eter is too low!");
        require(!hasAccess[msg.sender], "Already have a ticket");

        hasAccess[msg.sender] = true;
    }
}
```
