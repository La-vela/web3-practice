### ERC1155
예시
```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

// 상속
import "@openzepplin/contracts/token/ERC1155/ERC1155.sol";

contract RecipeGame is ERC1155 {
    // 아이템 ID 정의 (게임 아이템 관점)
    // 상수 관례 대문자
    uint256 public constant LEGENDARY_RECIPE =  1;     // 단 하나뿐인 레시피
    uint256 public constant COOK_BADGE = 2;     // 누구나 가질 수 있는 배지
    uint256 public constant INGEREDIENT_BOX = 3;    // 여러 개 가질 수 있는 재료 상자

    constructor() ERC1155("https://api.recipegame.com/metadata/{id}.json") {
        // 관리자에게 초기 아이템 지급 예시
        // ID 2번(배지)을 1000개 발행
        _mint(msg.sender, COOK_BADGE, 1000, "");
    }

    // 레시피 승인 시 NFT와 배지를 동시에 준다면?
    /**
     * @dev 레시피 승인 및 보상 지급
     * @param creator 레시피 제작자 주소
     * @param recipeId 레시피 고유 ID (이 자체가 NFT ID)
     */
    function approveAndReward(address creator, uint256 recipeId) public {
        // 1. 해당 레시피만의 고유 NFT 발행 (수량 1)
        _mint(creator, recipeId, 1, "");

        // 2. 참여 보상 배지 지급 (수량 5)
        _mint(creator, COOK_BADGE, 5, "");
    }


    // ※ 이 배지를 햄버거 게임에 100챌린지 증가 스코어로 활용할 수 있도록 
}


```
