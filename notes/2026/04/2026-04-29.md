### 메타버스 입장권 코드 
for Gemini

```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.17;

contract MetaverseGate {
    uint public immutable deadlineBlock; // 입장이 마감되는 블록 번호
    uint public constant TICKET_PRICE = 0.01 ether; // 입장료

    mapping(address => bool) public hasAccess; // bool맵핑 - 입장 권한 확인

    constructor(uint _durationBlocks) {
        // 현재 블록으로부터 특정 블록 수만큼만 판매 기간 설정
        deadlineBlock = block.number + _durationBlocks;
    }

    // [학습 내용 적용] 이더를 받고 입장권을 판매하는 함수
    function buyTicket() public payable {
        require(block.number <= deadlineBlock, "Registration closed!"); // [block.number] 활용
        require(msg.value >= TICKET_PRICE, "Not enough Ether sent!"); // [payable] 활용

        hasAccess[msg.sender] = true;
    }

    // 컨트랙트의 현재 잔액(수익금) 확인
    function getVaultBalance() public view returns (uint) {
        return address(this).balance;
    }
}
```
