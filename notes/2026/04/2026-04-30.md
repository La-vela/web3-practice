### 메타버스 입장권 확장 코드 (VIP 티켓)
for Gemini

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.17;

contract AdvancedMetaverseGate {
    address public owner;
    uint public immutable deadlineBlock;
    uint public constant TICKET_PRICE = 0.01 ether;
    uint public constant VIP_PRICE = 0.05 ether; // 더 비싼 VIP 티켓
    
    mapping(address => bool) public hasAccess;
    mapping(address => bool) public isVIP; // VIP 여부 추가
    uint public totalTicketsSold;

    event TicketPurchased(address indexed buyer, bool vip);

    constructor(uint _durationBlocks) {
        owner = msg.sender; // 컨트랙트 배포자를 주인으로 설정
        deadlineBlock = block.number + _durationBlocks;
    }

    // [획기적 기능 1] 일반/VIP 선택 구매
    function buyTicket(bool _wantVIP) public payable {
        require(block.number <= deadlineBlock, "Registration closed!");
        
        uint price = _wantVIP ? VIP_PRICE : TICKET_PRICE;
        require(msg.value >= price, "Check the ticket price!");

        hasAccess[msg.sender] = true;
        if (_wantVIP) {
            isVIP[msg.sender] = true;
        }
        
        totalTicketsSold++;
        emit TicketPurchased(msg.sender, _wantVIP);
    }

    // [획기적 기능 2] 수익금 인출 (Owner 전용)
    // 아무나 돈을 가져가지 못하게 관리자 권한을 부여합니다.
    function withdraw() public {
        require(msg.sender == owner, "Only owner can withdraw");
        payable(owner).transfer(address(this).balance);
    }

    // [획기적 기능 3] 남은 시간(블록) 확인
    function getBlocksLeft() public view returns (uint) {
        if (block.number >= deadlineBlock) return 0;
        return deadlineBlock - block.number;
    }
}
